 39:     /**
 40:      * Debugs a variable
 41:      * Only visible to admins if WP_DEBUG is on
 42:      * @param mixed $var the var to debug
 43:      * @param bool $die (optional) whether to die after outputting
 44:      * @param string $function (optional) the function to call, usually either print_r or var_dump, but can be anything
 45:      * @return unknown
 46:      */
 47:     function log( $var, $die = false, $function = 'var_dump' ) {
 48:         if ( !current_user_can( 'manage_options' ) || !WP_DEBUG ) {
 49:             return;
 50:         }
 51: 
 52:         ob_start();
 53:         if ( is_string( $var ) ) {
 54:             echo "- ".$var."\n";
 55:         } else {
 56:             call_user_func( $function, $var );
 57:         }
 58: 
 59:         if ( $die ) {
 60:             die();
 61:         }
 62: 
 63:         $debug = ob_get_clean();
 64: 
 65:         $this->history[] = $debug;
 66: 
 67:         //allow this to be used as a filter
 68:         return $var;
 69:     }
